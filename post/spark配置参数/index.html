Spark 配置参数

#配置 Storage 内存区域大小，默认为 0.6，即 60%
spark.storage.memoryFraction=0.6 

#Executor 堆外内存，默认 300GB，需要在 Spark 的启动脚本中配置
spark.executor.memoryOverhead=300 

#Executor 的 BlockManager 在拉取其他节点上 BlockManager 数据时候的最大等待时间，默认为 120s，需要在 Spark 的启动脚本中配置
spark.core.connection.ack.wait.timeout=120
